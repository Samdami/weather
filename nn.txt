function applyConfig(config) {
  const bg = config.background_color || defaultConfig.background_color;
  const surface = config.surface_color || defaultConfig.surface_color;
  const text = config.text_color || defaultConfig.text_color;
  const primary = config.primary_action_color || defaultConfig.primary_action_color;
  const secondary = config.secondary_action_color || defaultConfig.secondary_action_color;
  const font = config.font_family || defaultConfig.font_family;
  const size = config.font_size || defaultConfig.font_size;

  const isDark = document.documentElement.classList.contains('dark');

  if (!isDark) {
    document.body.style.backgroundColor = bg;
    document.body.style.color = text;
    document.querySelectorAll('.rounded-2xl').forEach(el => {
      if (el.classList.contains('bg-white')) el.style.backgroundColor = surface;
    });
  }

  const title = config.app_title || defaultConfig.app_title;
  const el = document.getElementById('app-logo-text');
  if (el) el.textContent = title;

  document.body.style.fontFamily = `${font}, DM Sans, system-ui, sans-serif`;

  const baseSize = size;
  document.querySelectorAll('#hero-location, .text-sm').forEach(e => e.style.fontSize = `${baseSize}px`);
  document.getElementById('hero-temp').style.fontSize = `${baseSize * 5}px`;
}

function toF(c) { return Math.round(c * 9/5 + 32); }
function displayTemp(c) { return useCelsius ? Math.round(c) + '°' : toF(c) + '°'; }
function displayWind(kmh) { return useCelsius ? kmh + ' km/h' : Math.round(kmh * 0.621) + ' mph'; }

function toggleUnit() {
  useCelsius = !useCelsius;
  document.getElementById('unit-toggle').textContent = useCelsius ? '°C' : '°F';
  renderCity(currentCity);
}

function toggleTheme() {
  document.documentElement.classList.toggle('dark');
  renderCharts(CITIES[currentCity]);
}

function renderCity(name) {
  currentCity = name;
  const d = CITIES[name];
  if (!d) return;

  document.getElementById('hero-location').textContent = name;
  document.getElementById('hero-temp').textContent = displayTemp(d.temp);
  document.getElementById('hero-condition').textContent = d.condition;
  document.getElementById('hero-hilo').textContent = `H:${displayTemp(d.hi)} L:${displayTemp(d.lo)}`;
  document.getElementById('hero-icon').textContent = d.icon;

  const now = new Date();
  document.getElementById('hero-datetime').textContent = now.toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric' }) + ' · ' + now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' });

  document.getElementById('card-humidity').innerHTML = d.humidity + '<span class="text-sm font-normal">%</span>';
  document.getElementById('card-wind').innerHTML = displayWind(d.wind);
  document.getElementById('card-wind-dir').textContent = `${d.windDir} gusts ${displayWind(d.gust)}`;
  document.getElementById('card-rain').innerHTML = d.rain + ' <span class="text-sm font-normal">mm</span>';
  document.getElementById('card-pressure').innerHTML = d.pressure + ' <span class="text-sm font-normal">hPa</span>';

  const alertBanner = document.getElementById('alert-banner');
  const cfg = window.elementSdk.config;
  const customAlert = cfg.alert_message;
  if (d.alert || customAlert) {
    alertBanner.classList.remove('hidden');
    document.getElementById('alert-text').textContent = customAlert || d.alertMsg;
  } else {
    alertBanner.classList.add('hidden');
  }

  renderForecast(d);
  renderCharts(d);
  renderMetarTaf(d);
}

function renderForecast(d) {
  const grid = document.getElementById('forecast-grid');
  grid.innerHTML = '';
  const today = new Date().getDay();
  for (let i = 0; i < 7; i++) {
    const dayIdx = (today + i) % 7;
    const hi = d.hi + Math.round((Math.random() - 0.5) * 6);
    const lo = d.lo + Math.round((Math.random() - 0.5) * 4);
    const iconIdx = i === 0 ? FORECAST_ICONS.indexOf(d.icon) !== -1 ? FORECAST_ICONS.indexOf(d.icon) : 0 : Math.floor(Math.random() * FORECAST_ICONS.length);
    const col = document.createElement('div');
    col.className = 'flex flex-col items-center gap-2 py-3 rounded-xl hover:bg-stone-50 dark:hover:bg-white/[0.03] transition-colors';
    col.innerHTML = `
      <span class="text-xs font-medium text-stone-500 dark:text-stone-400">${i === 0 ? 'Today' : DAYS[dayIdx]}</span>
      <span class="text-2xl">${FORECAST_ICONS[iconIdx]}</span>
      <span class="text-sm font-semibold font-mono">${displayTemp(hi)}</span>
      <span class="text-xs text-stone-400 dark:text-stone-500 font-mono">${displayTemp(lo)}</span>
    `;
    grid.appendChild(col);
  }
}

// Search
const searchInput = document.getElementById('search-input');
const searchDropdown = document.getElementById('search-dropdown');
const cityNames = Object.keys(CITIES);

searchInput.addEventListener('input', () => {
  const q = searchInput.value.trim().toLowerCase();
  if (!q) { searchDropdown.classList.add('hidden'); return; }
  const matches = cityNames.filter(c => c.toLowerCase().includes(q));
  if (!matches.length) {
    searchDropdown.innerHTML = '<div class="px-4 py-3 text-sm text-stone-400">No cities found</div>';
  } else {
    searchDropdown.innerHTML = matches.map(c => `<button onclick="selectCity('${c}')" class="w-full text-left px-4 py-2.5 text-sm hover:bg-stone-50 dark:hover:bg-white/5 transition-colors flex items-center gap-2"><i data-lucide="map-pin" style="width:13px;height:13px;opacity:0.4"></i>${c}</button>`).join('');
  }
  searchDropdown.classList.remove('hidden');
  lucide.createIcons();
});

searchInput.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    const q = searchInput.value.trim().toLowerCase();
    const match = cityNames.find(c => c.toLowerCase().startsWith(q));
    if (match) selectCity(match);
  }
});

document.addEventListener('click', (e) => {
  if (!e.target.closest('#search-input') && !e.target.closest('#search-dropdown')) {
    searchDropdown.classList.add('hidden');
  }
});

function selectCity(name) {
  searchInput.value = '';
  searchDropdown.classList.add('hidden');
  renderCity(name);
}
